

CREATE TABLE diet_cycles (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    code VARCHAR(50) UNIQUE NOT NULL,
    name_en VARCHAR(255) NOT NULL,
    name_si VARCHAR(255),
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);


DROP TABLE IF EXISTS norm_weights;

CREATE TABLE norm_weights (
    item_id INTEGER NOT NULL REFERENCES items(id) ON DELETE CASCADE,
    diet_type_id INTEGER NOT NULL REFERENCES diet_types(id) ON DELETE CASCADE, -- Matched to your SERIAL id!
    meal VARCHAR(20) NOT NULL,
    weight DECIMAL(10,2) DEFAULT 0,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    
    PRIMARY KEY (item_id, diet_type_id, meal) 
);

 
-- Daily calculation snapshots (one run per date)
CREATE TABLE IF NOT EXISTS calculation_runs (
    id SERIAL PRIMARY KEY,
    calc_date DATE NOT NULL UNIQUE,
    patient_cycle VARCHAR(100) NOT NULL,
    staff_cycle VARCHAR(100) NOT NULL,
 
    -- Aggregated patient totals (snapshot)
    patient_totals JSONB NOT NULL DEFAULT '{}'::jsonb,
    -- e.g. {"NOR": 47, "DM": 24, "S1": 15, "S2": 5, "S3": 2, ...}
 
    staff_breakfast INTEGER NOT NULL DEFAULT 0,
    staff_lunch INTEGER NOT NULL DEFAULT 0,
    staff_dinner INTEGER NOT NULL DEFAULT 0,
 
    -- Special request totals (hospital-wide)   
    soup_count INTEGER NOT NULL DEFAULT 0,
    kanda_count INTEGER NOT NULL DEFAULT 0,
    pol_sambola_count INTEGER NOT NULL DEFAULT 0,
 
    -- Extra items aggregated (raw sums across wards)
    extras_totals JSONB NOT NULL DEFAULT '{}'::jsonb,
    custom_extras_totals JSONB NOT NULL DEFAULT '[]'::jsonb,
 
    status VARCHAR(30) NOT NULL DEFAULT 'calculated',
    -- calculated → draft_po → pending_approval → approved → invoiced
 
    calculated_by INTEGER REFERENCES users(id),
    calculated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
 
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);
 
-- Individual calculation line items (one per item × meal)
CREATE TABLE IF NOT EXISTS calculation_line_items (
    id SERIAL PRIMARY KEY,
    calc_run_id INTEGER NOT NULL REFERENCES calculation_runs(id) ON DELETE CASCADE,
    item_id INTEGER NOT NULL REFERENCES items(id),
    meal VARCHAR(20) NOT NULL, -- 'breakfast', 'lunch', 'dinner', 'extra', 'recipe'
 
    -- Per-diet-type breakdown (grams in base unit)
    breakdown JSONB NOT NULL DEFAULT '{}'::jsonb,
    -- e.g. {"NOR": {"count": 47, "norm_g": 75, "total_g": 3525}, "DM": {...}, ...}
 
    subtotal_base NUMERIC(14,2) NOT NULL DEFAULT 0,  -- Total in base unit (g, ml, pcs)
    base_unit VARCHAR(10) NOT NULL DEFAULT 'g',       -- 'g', 'ml', 'pcs'
 
    display_value NUMERIC(14,4) NOT NULL DEFAULT 0,   -- Converted to display unit
    display_unit VARCHAR(30) NOT NULL DEFAULT 'Kg',    -- Item's preferred unit
 
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
 
    CONSTRAINT uq_calc_item_meal UNIQUE (calc_run_id, item_id, meal)
);
 
-- Vegetable category summaries (for allocation UI)
CREATE TABLE IF NOT EXISTS calculation_veg_summaries (
    id SERIAL PRIMARY KEY,
    calc_run_id INTEGER NOT NULL REFERENCES calculation_runs(id) ON DELETE CASCADE,
    veg_category VARCHAR(30) NOT NULL, -- 'palaa', 'gedi', 'piti', 'other'
    required_kg NUMERIC(12,4) NOT NULL DEFAULT 0,
 
    CONSTRAINT uq_calc_veg_cat UNIQUE (calc_run_id, veg_category)
);
 
-- Purchase order line items (generated from calculation)
CREATE TABLE IF NOT EXISTS po_line_items (
    id SERIAL PRIMARY KEY,
    calc_run_id INTEGER NOT NULL REFERENCES calculation_runs(id) ON DELETE CASCADE,
    item_id INTEGER NOT NULL REFERENCES items(id),
    category_id INTEGER NOT NULL,
 
    calculated_qty NUMERIC(14,4) NOT NULL DEFAULT 0,
    display_unit VARCHAR(30) NOT NULL,
    unit_price NUMERIC(12,2) NOT NULL DEFAULT 0,
    total_price NUMERIC(14,2) NOT NULL DEFAULT 0,
 
    -- Meal flags
    for_breakfast BOOLEAN DEFAULT FALSE,
    for_lunch BOOLEAN DEFAULT FALSE,
    for_dinner BOOLEAN DEFAULT FALSE,
    for_extra BOOLEAN DEFAULT FALSE,
    for_kanda BOOLEAN DEFAULT FALSE,
 
    -- Accountant revision
    revised_qty NUMERIC(14,4),
    revised_price NUMERIC(12,2),
    revised_total NUMERIC(14,2),
 
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
 
    CONSTRAINT uq_po_item UNIQUE (calc_run_id, item_id)
);
 
-- Recipe calculation results (for kitchen cook sheet)
CREATE TABLE IF NOT EXISTS calculation_recipe_results (
    id SERIAL PRIMARY KEY,
    calc_run_id INTEGER NOT NULL REFERENCES calculation_runs(id) ON DELETE CASCADE,
    recipe_id INTEGER NOT NULL REFERENCES recipes(id),
    patient_count INTEGER NOT NULL DEFAULT 0,       -- How many patients requested this
    weighted_count NUMERIC(10,2) NOT NULL DEFAULT 0, -- After diet conversion factors
 
    -- Calculated ingredient quantities
    ingredients JSONB NOT NULL DEFAULT '[]'::jsonb,
    -- e.g. [{"itemId": 5, "nameEn": "Coconut", "nameSi": "පොල්", "qty": 9.6, "unit": "seeds"}, ...]
 
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
 
    CONSTRAINT uq_calc_recipe UNIQUE (calc_run_id, recipe_id)
);
 
-- Diet conversion factors for recipes (Pol Sambola, Soup, Kanda)
-- Configurable by Hospital Admin via Recipe Editor
CREATE TABLE IF NOT EXISTS recipe_diet_factors (
    id SERIAL PRIMARY KEY,
    recipe_id INTEGER NOT NULL REFERENCES recipes(id) ON DELETE CASCADE,
    diet_type_id INTEGER NOT NULL REFERENCES diet_types(id) ON DELETE CASCADE,
    factor NUMERIC(6,4) NOT NULL DEFAULT 1.0,
 
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
 
    CONSTRAINT uq_recipe_diet_factor UNIQUE (recipe_id, diet_type_id)
);

  -- 1. Purchase Orders header table
CREATE TABLE purchase_orders (
    id              SERIAL PRIMARY KEY,
    calc_run_id     INTEGER NOT NULL REFERENCES calculation_runs(id),
    po_date         DATE NOT NULL,
    bill_number     VARCHAR(20),
    status          VARCHAR(30) NOT NULL DEFAULT 'draft',
    original_total  NUMERIC(14,2) NOT NULL DEFAULT 0,
    revised_total   NUMERIC(14,2),
    created_by      INTEGER REFERENCES users(id),
    submitted_at    TIMESTAMPTZ,
    submitted_by    INTEGER REFERENCES users(id),
    reviewed_at     TIMESTAMPTZ,
    reviewed_by     INTEGER REFERENCES users(id),
    rejection_reason TEXT,
    created_at      TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at      TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT uq_po_date UNIQUE (po_date)
);

-- 2. Purchase Order Line Items
CREATE TABLE po_items (
    id              SERIAL PRIMARY KEY,
    po_id           INTEGER NOT NULL REFERENCES purchase_orders(id) ON DELETE CASCADE,
    item_id         INTEGER NOT NULL REFERENCES items(id),
    category_id     INTEGER NOT NULL REFERENCES categories(id),
    quantity        NUMERIC(14,4) NOT NULL DEFAULT 0,
    unit            VARCHAR(30) NOT NULL,
    unit_price      NUMERIC(12,2) NOT NULL DEFAULT 0,
    default_price   NUMERIC(12,2) NOT NULL DEFAULT 0,
    total_price     NUMERIC(14,2) NOT NULL DEFAULT 0,
    is_price_changed BOOLEAN DEFAULT FALSE,
    for_breakfast   BOOLEAN DEFAULT FALSE,
    for_lunch       BOOLEAN DEFAULT FALSE,
    for_dinner      BOOLEAN DEFAULT FALSE,
    for_extra       BOOLEAN DEFAULT FALSE,
    for_kanda       BOOLEAN DEFAULT FALSE,
    revised_qty     NUMERIC(14,4),
    revised_price   NUMERIC(12,2),
    revised_total   NUMERIC(14,2),
    created_at      TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT uq_po_item_unique UNIQUE (po_id, item_id)
);

-- 1. The main Invoices table
CREATE TABLE invoices (
    id SERIAL PRIMARY KEY,
    po_id INTEGER NOT NULL REFERENCES purchase_orders(id) ON DELETE CASCADE,
    
    invoice_number VARCHAR(100) NOT NULL, -- The number printed on the supplier's paper bill
    invoice_date DATE NOT NULL,
    
    status VARCHAR(30) NOT NULL DEFAULT 'pending_receipt', 
    -- Workflow: 'pending_receipt' (waiting at back door) -> 'received' (kitchen checked) -> 'paid' (accountant paid)
    
    billed_total NUMERIC(14,2) DEFAULT 0, -- The total amount calculated after receiving
    
    received_by INTEGER REFERENCES users(id), -- Tracks WHICH kitchen staff member accepted the delivery
    received_at TIMESTAMPTZ,
    
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 2. The Invoice Items (The Kitchen's Checklist)
CREATE TABLE invoice_items (
    id SERIAL PRIMARY KEY,
    invoice_id INTEGER NOT NULL REFERENCES invoices(id) ON DELETE CASCADE,
    po_item_id INTEGER NOT NULL REFERENCES po_items(id), -- Links directly to what was ordered
    item_id INTEGER NOT NULL REFERENCES items(id),
    
    ordered_qty NUMERIC(14,4) NOT NULL DEFAULT 0,
    received_qty NUMERIC(14,4) NOT NULL DEFAULT 0, -- Kitchen will fill this in!
    
    unit_price NUMERIC(12,2) NOT NULL DEFAULT 0,
    total_price NUMERIC(14,2) NOT NULL DEFAULT 0, -- received_qty * unit_price
    
    status VARCHAR(30) NOT NULL DEFAULT 'pending',
    -- 'pending', 'complete' (got all of it), 'partial' (missing some), 'rejected' (bad quality)
    
    notes TEXT, -- e.g., "Rejected 2kg of tomatoes because they were spoiled"
    
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

-- 3. Indexes for fast loading
CREATE INDEX idx_invoices_po_id ON invoices(po_id);
CREATE INDEX idx_invoices_status ON invoices(status);
CREATE INDEX idx_invoice_items_invoice_id ON invoice_items(invoice_id);
