CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    full_name VARCHAR(150) NOT NULL,
    email VARCHAR(150) NOT NULL UNIQUE,
    password_hash TEXT NOT NULL,
    role VARCHAR(30) NOT NULL CHECK (
        role IN (
            'SYSTEM_ADMIN',
            'HOSPITAL_ADMIN',
            'DIET_CLERK',
            'SUBJECT_CLERK',
            'ACCOUNTANT',
            'KITCHEN'
        )
    ),
    is_active BOOLEAN NOT NULL DEFAULT TRUE,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);



CREATE TABLE audit_logs (
    id BIGSERIAL PRIMARY KEY,

    timestamp TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,

    user_id BIGINT,
    user_name VARCHAR(255),
    user_role VARCHAR(100),

    action VARCHAR(150) NOT NULL,
    entity VARCHAR(150),

    entity_id VARCHAR(100),

    old_value JSONB,
    new_value JSONB,

    details JSONB,

    severity VARCHAR(50) DEFAULT 'info',

    status_code INTEGER,
    success BOOLEAN DEFAULT TRUE
);


CREATE TABLE wards (
    id BIGSERIAL PRIMARY KEY,
    ward_code VARCHAR(50) NOT NULL UNIQUE,
    ward_name VARCHAR(150) NOT NULL,
    bed_count INTEGER NOT NULL DEFAULT 0,
    cot_count INTEGER NOT NULL DEFAULT 0,
    icu_count INTEGER NOT NULL DEFAULT 0,
    incubator_count INTEGER NOT NULL DEFAULT 0,
    is_active BOOLEAN NOT NULL DEFAULT TRUE,
    created_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMPTZ DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE items (
    id SERIAL PRIMARY KEY,
    name_en VARCHAR(150) NOT NULL,
    name_si VARCHAR(150) NOT NULL,
    unit VARCHAR(50) DEFAULT 'Kg',
    default_price NUMERIC(10,2) DEFAULT 0,

    category_id INTEGER NOT NULL,

    is_protein BOOLEAN DEFAULT FALSE,
    diet_cycle VARCHAR(50),

    is_vegetable BOOLEAN DEFAULT FALSE,
    veg_category VARCHAR(50),

    is_extra BOOLEAN DEFAULT FALSE,
    calc_type VARCHAR(50) DEFAULT 'norm_weight',

    active BOOLEAN DEFAULT TRUE,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE TABLE recipes (
    id SERIAL PRIMARY KEY,
    recipe_key VARCHAR(100) UNIQUE NOT NULL,
    name VARCHAR(150) NOT NULL,
    active BOOLEAN NOT NULL DEFAULT TRUE,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE recipe_ingredients (
    id SERIAL PRIMARY KEY,
    recipe_id INTEGER NOT NULL REFERENCES recipes(id) ON DELETE CASCADE,
    item_id INTEGER NOT NULL REFERENCES items(id) ON DELETE RESTRICT,
    norm_per_patient NUMERIC(10,2) NOT NULL DEFAULT 0,
    unit VARCHAR(50) NOT NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT uq_recipe_item UNIQUE (recipe_id, item_id)
);

CREATE TABLE census_entries (
    id SERIAL PRIMARY KEY,
    ward_id INTEGER NOT NULL,
    entry_date DATE NOT NULL,
    status VARCHAR(30) NOT NULL DEFAULT 'draft',
    total_patients INTEGER NOT NULL DEFAULT 0,
    diets JSONB NOT NULL DEFAULT '{}'::jsonb,
    special JSONB NOT NULL DEFAULT '{}'::jsonb,
    extras JSONB NOT NULL DEFAULT '{}'::jsonb,
    custom_extras JSONB NOT NULL DEFAULT '[]'::jsonb,
    submitted_by INTEGER NULL,
    submitted_at TIMESTAMP NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    CONSTRAINT uq_census_ward_date UNIQUE (ward_id, entry_date)
);
CREATE TABLE staff_meals (
    id SERIAL PRIMARY KEY,
    meal_date DATE NOT NULL UNIQUE,
    breakfast INTEGER NOT NULL DEFAULT 0,
    lunch INTEGER NOT NULL DEFAULT 0,
    dinner INTEGER NOT NULL DEFAULT 0,
    staff_cycle VARCHAR(100) DEFAULT 'Chicken',
    status VARCHAR(30) NOT NULL DEFAULT 'submitted',
    submitted_by INTEGER NULL,
    submitted_at TIMESTAMP NULL,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);
